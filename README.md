# Simple Math Function Example

def square(number):
    """
    Returns the square of a given number.
    
    Args:
        number (int or float): The number to square.
        
    Returns:
        int or float: The squared result.
    """
    return number ** 2  # Use exponent operator to square the number


# Example usage
num = 5
result = square(num)
print(f"The square of {num} is {result}")
